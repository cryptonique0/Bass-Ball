```typescript
import { discoverTokens, getTokenPrice } from './lib/web3/base-ecosystem';

const tokens = await discoverTokens(8453); // Base chain
const price = await getTokenPrice(tokens[0].address);
console.log(`Token price: ${price}`);
```

### Cross-Chain Transfer
```typescript
import { compareRoutes, executeBridge } from './lib/web3/cross-chain-swap-optimizer';

const routes = await compareRoutes(1, 8453); // Ethereum to Base
const bestRoute = routes[0]; // Cheapest/fastest
const tx = await executeBridge(bestRoute);
```

### Create Proposal
```typescript
import { createProposal, vote } from './lib/web3/governance-dao';

const proposal = await createProposal({
  title: 'Increase LP rewards',
  description: '...',
  options: ['Yes', 'No', 'Abstain']
});

await vote(proposal.id, 0); // Vote Yes
```

### Bridge NFT
```typescript
import { createWormholeNFTBridgeRequest, getWormholeBridgeStatus } from './lib/web3/nft-bridge';

const request = createWormholeNFTBridgeRequest(nft, 8453, recipient);
```
